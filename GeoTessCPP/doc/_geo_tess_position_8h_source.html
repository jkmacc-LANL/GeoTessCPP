213  void updateRadialCoefficients()
214  {
215  // make sure dimensions of radialIndexes and radialCoefficients
216  // are at least as big as the number of vertices involved in interpolation.
217  if (radialIndexes.size() < vertices[tessid].size())
218  {
219  radialIndexes.resize(vertices[tessid].size());
220  radialCoefficients.resize(vertices[tessid].size());
221  }
222 
223  // if radial interpolation coefficients have already been computed
224  // do nothing.
225  if (radialIndexes[0].size() == 0)
226  {
227  vector<int>& v = vertices[tessid];
228  for (int i = 0; i < (int)v.size(); ++i)
229  modlProfiles[v[i]][layerId]->setInterpolationCoefficients(radialInterpolatorType,
230  radialIndexes[i], radialCoefficients[i], radius, radiusOutOfRangeAllowed);
231  }
232  }

260  int biggerTriangle(int t1, int t2)
261  {
262  const int* tv = grid.getTriangleVertexIndexes(t1);
263  double dot1 = GeoTessUtils::dot(grid.getVertex(tv[0]), grid.getVertex(tv[1]))
264  + GeoTessUtils::dot(grid.getVertex(tv[1]), grid.getVertex(tv[2]))
265  + GeoTessUtils::dot(grid.getVertex(tv[2]), grid.getVertex(tv[0]));
266 
267  tv = grid.getTriangleVertexIndexes(t2);
268  double dot2 = GeoTessUtils::dot(grid.getVertex(tv[0]), grid.getVertex(tv[1]))
269  + GeoTessUtils::dot(grid.getVertex(tv[1]), grid.getVertex(tv[2]))
270  + GeoTessUtils::dot(grid.getVertex(tv[2]), grid.getVertex(tv[0]));
271 
272  return dot2 > dot1 ? t1 : t2;
273
