316  LONG_INT getMemory()
317  {
318  LONG_INT memory = (LONG_INT) sizeof(GeoTessMetaData);
319 
320  // EarthShape earthShape;
321  memory += (LONG_INT) sizeof(EarthShape);
322 
323  memory += (LONG_INT) (description.length() + attributeFilterString.length()
324  + inputModelFile.length() + inputGridFile.length() + outputModelFile.length()
325  + outputGridFile.length() + modelSoftwareVersion.length() + modelGenerationDate.length());
326 
327  // string* layerNames;
328  memory += nLayers * (LONG_INT) sizeof(string);
329  for (int i=0; i<nLayers; ++i)
330  memory += (LONG_INT) layerNames[i].length();
331 
332  // int* layerTessIds;
333  memory += nLayers * (LONG_INT)sizeof(int);
334 
335  // GeoTessDataType* dataType;
336  memory += nLayers * (int)sizeof(GeoTessDataType);
337 
338  // string* attributeNames;
339  // string* attributeUnits;
340  if (nAttributes > 0)
341  {
342  memory += 2 * nAttributes * (LONG_INT) sizeof(string);
343  for (int i=0; i<nAttributes; ++i)
344  {
345  memory += (LONG_INT) attributeNames[i].length();
346  memory += (LONG_INT) attributeUnits[i].length();
347  }
348  }
349 
350  // vector<int> attributeFilter;
351  memory += (LONG_INT) (attributeFilter.capacity() * sizeof(int));
352 
353  // vector<int> inputFilter;
354  memory += (LONG_INT) (inputFilter.capacity() * sizeof(int));
355 
356  return memory;
357  }

811  void setAttributes(const string& nms, const string& unts)
812  {
813  vector<string> names, units;
814  CPPUtils::tokenizeString(nms, ";", names);
815  CPPUtils::tokenizeString(unts, ";", units);
816  if (names.size() != units.size())
817  {
818  ostringstream os;
819  os << "Error in GeoTessMetaData::setAttributes(const string& nms, const string& unts)" << endl
820  << "Attribute names size (" << names.size()
821  << ") is not equal to units size (" << units.size() << ")" << endl;
822  names.clear(); units.clear();
823  throw GeoTessException(os, __FILE__, __LINE__, 6009);
824  }
825  setAttributes(names, units);
826 
827  properties["attributeNames"] = nms;
828  properties["attributeUnits"] = unts;
829  }
